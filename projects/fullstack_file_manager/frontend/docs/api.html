分类编辑

  • type:post
  • url:/api/category/edit

  • input

    {
    new:[{
        name:'ss',
        description:'ss',
        parent_id:1, // 如果没有 parent_id 的话，会默认是第一类
        children:[{
            name:'ss',
            description:'abc', // 这种情况下，默认上一级是自己的 parent_id
        }]
    }],
    del:[1,2,3] // 被删除了的分类,
    // 更新的分类
    update:[{
        id:1,
        name:'sss',
        description:'11sss'
    },{
        id:1,
        name:'sss',
        description:'11sss'
    }]
}
